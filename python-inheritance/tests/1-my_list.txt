# Test Cases for MyList Class
# ===========================

# Test 1: Basic functionality - empty list
test1 = MyList()
test1.print_sorted()  # Expected: []

# Test 2: Basic functionality - already sorted list
test2 = MyList([1, 2, 3, 4, 5])
test2.print_sorted()  # Expected: [1, 2, 3, 4, 5]
print(test2)          # Expected: [1, 2, 3, 4, 5] (unchanged)

# Test 3: Basic functionality - reverse sorted list
test3 = MyList([5, 4, 3, 2, 1])
test3.print_sorted()  # Expected: [1, 2, 3, 4, 5]
print(test3)          # Expected: [5, 4, 3, 2, 1] (unchanged)

# Test 4: Basic functionality - random order
test4 = MyList([3, 1, 4, 1, 5, 9, 2, 6])
test4.print_sorted()  # Expected: [1, 1, 2, 3, 4, 5, 6, 9]
print(test4)          # Expected: [3, 1, 4, 1, 5, 9, 2, 6] (unchanged)

# Test 5: Negative numbers
test5 = MyList([-5, 0, 5, -10, 10])
test5.print_sorted()  # Expected: [-10, -5, 0, 5, 10]
print(test5)          # Expected: [-5, 0, 5, -10, 10] (unchanged)

# Test 6: Single element
test6 = MyList([42])
test6.print_sorted()  # Expected: [42]
print(test6)          # Expected: [42] (unchanged)

# Test 7: Duplicate elements
test7 = MyList([7, 7, 7, 7])
test7.print_sorted()  # Expected: [7, 7, 7, 7]
print(test7)          # Expected: [7, 7, 7, 7] (unchanged)

# Test 8: Using append() method
test8 = MyList()
test8.append(5)
test8.append(2)
test8.append(8)
test8.print_sorted()  # Expected: [2, 5, 8]
print(test8)          # Expected: [5, 2, 8] (unchanged)

# Test 9: Inheritance - check if MyList has all list methods
test9 = MyList([1, 2, 3])
print(isinstance(test9, list))  # Expected: True
print(len(test9))               # Expected: 3
test9.append(4)
print(test9)                    # Expected: [1, 2, 3, 4]
test9.pop()
print(test9)                    # Expected: [1, 2, 3]

# Test 10: Multiple print_sorted calls
test10 = MyList([3, 1, 2])
test10.print_sorted()  # Expected: [1, 2, 3]
test10.print_sorted()  # Expected: [1, 2, 3] (again)
print(test10)          # Expected: [3, 1, 2] (still unchanged)
